def uxt(x, t):
    if t == 0:
        return 1
    if x == 0 or x == 1:
        return 0
    C = 4 / np.pi
    
    m = 0
    S = 0
    while m <= 100000:
        M = 2 * m + 1
        sin = np.sin(np.pi * M * x)
        m += 1
        exp = np.exp(-(np.pi ** 2) * (M ** 2) * t)
        S += (exp / M) * sin
    S *= C
    return S
